74  SeqHW(SeqManager * o,
75  unsigned i,
76  volatile void *ctrl,

78  :owner(o)

80  ,ctrlreg(ctrl)
81  ,rambase(ram)
82  ,loaded(0)
83  ,running(false)
84  ,ctrlreg_user(0u)
85  ,ctrlreg_hw(0u)
86  {
87  switch(owner->type) {
88  case SeqManager::TypeEVG:
89  ctrlreg_user |= 31;
90  break;
91  case SeqManager::TypeEVR:
92  ctrlreg_user |= 63;
93  break;
94  default:
95  return;
96  }
97  ctrlreg_hw = ctrlreg_user;
98 
99  nat_iowrite32(ctrlreg, ctrlreg_hw | EVG_SEQ_RAM_RESET);
100  }
101 
102  // call with interruptLock
103  void arm()
104  {
105  ctrlreg_hw = ctrlreg_user;
106  nat_iowrite32(ctrlreg, ctrlreg_hw | EVG_SEQ_RAM_ARM);
107  }
108 
109  // call with interruptLock
110  bool disarm()
111  {
112 
113  // EVG_SEQ_RAM_DISABLE is really "stop"
114  // to avoid aborting a potentially running sequencer, switch the trigger source to disable
115 
116  ctrlreg_hw &= ~(EVG_SEQ_RAM_SRC_MASK);
117 
118  switch(owner->type) {
119  case SeqManager::TypeEVG:
120  ctrlreg_hw |= 31;
121  break;
122  case SeqManager::TypeEVR:
123  ctrlreg_hw |= 63;
124  break;
125  }
126 
127  nat_iowrite32(ctrlreg, ctrlreg_hw);
128  bool isrun = nat_ioread32(ctrlreg) & EVG_SEQ_RAM_RUNNING;
129 
130  return isrun;
131  }

183  void setTimestamp(const double *arr, epicsUInt32 count)
184  {
185  const double tmult = getTimeScale();
186  times_t times(count);
187  // check for monotonic
188  // TODO: not handling overflow (HW supports controlled rollover w/ special 0xffffffff times)
189  for(epicsUInt32 i=0; i<count; i++)
190  {
191  if(!finite(arr[i]) || arr[i]<0.0) {
192  std::string msg("Times must be finite >=0");
193  last_err = msg;
194  scanIoRequest(onErr);
195  throw std::runtime_error(msg);
196  }
197 
198  times[i] = (arr[i]*tmult)+0.5;
199 
200  if(i>0 && times[i]<=times[i-1]) {
201  std::string msg("Non-monotonic timestamp array");
202  last_err = msg;
203  scanIoRequest(onErr);
204  throw std::runtime_error(msg);
205  } else if(times[i]==0xffffffff) {
206  std::string msg("Time overflow, rollover not supported");
207  last_err = msg;
208  scanIoRequest(onErr);
209  throw std::runtime_error(msg);
210  }
211  }
212  {
213  SCOPED_LOCK(mutex);
214  scratch.times.swap(times);
215  is_committed = false;
216  }
217  DEBUG(4, ("Set times\n"));
218  scanIoRequest(changed);
219  }

438 void SoftSequence::load()
439 {
440  SCOPED_LOCK(mutex);
441  DEBUG(3, ("Loading %c\n", hw ? 'L' : 'U') );
442  if(hw) {DEBUG(3, ("Skip\n")); return;}
443 
444  // find unused SeqHW
445  {
446  interruptLock L;
447 
448  is_insync = false; // paranoia
449 
450  for(size_t i=0, N=owner->hw.size(); i<N; i++) {
451  SeqHW *temp = owner->hw[i];
452  if(temp && !temp->loaded) {
453  temp->loaded = this;
454  hw = temp;
455  break;
456  }
457  }
458 
459  if(hw) {
460  // paranoia: disable any external trigger mappings
461  owner->mapTriggerSrc(hw->idx, 0x02000000);
462 
463  if(!hw->disarm())
464  sync();
465  }
466  }
467 
468  if(!hw) {
469  last_err = "All HW Seq. in use";
470  scanIoRequest(onErr);
471  throw alarm_exception(MAJOR_ALARM, WRITE_ALARM);
472  }
473 
474  // clear residual error (if any)
475  last_err = "";
476  scanIoRequest(onErr);
477 
478  scanIoRequest(changed);
479  DEBUG(1, ("Loaded\n") );
480 }

507 void SoftSequence::commit()
508 {
509  SCOPED_LOCK(mutex);
510  DEBUG(3, ("Committing %c\n", is_committed ? 'Y' : 'N') );
511 
512  if(is_committed) {DEBUG(3, ("Skip\n")); return;}
513 
514  // scratch.times already check for monotonic
515 
516  Config conf(scratch); // vector copies
517 
518  size_t buflen = std::min(conf.codes.size(),
519  conf.times.size());
520  conf.codes.resize(buflen);
521  conf.times.resize(buflen);
522 
523  // ensure presence of trailing end of sequence marker event 0x7f
524  if(conf.codes.empty() || conf.codes.back()!=0x7f)
525  {
526  if(!conf.times.empty() && conf.times.back()==0xffffffff) {
527  std::string msg("Input array is missing 0x7f and maxing out times");
528  last_err = msg;
529  scanIoRequest(onErr);
530  throw std::runtime_error(msg);
531  }
532 
533  conf.codes.push_back(0x7f);
534 
535  if(conf.times.empty())
536  conf.times.push_back(0);
537  else
538  conf.times.push_back(conf.times.back()+1);
539  }
540 
541  if(conf.times.size()>2048) {
542  std::string msg("Sequence too long");
543  last_err = msg;
544  scanIoRequest(onErr);
545  throw std::runtime_error(msg);
546  }
547 
548  assert(!hw || hw->loaded==this);
549 
550  {
551  interruptLock L;
552  committed.swap(conf);
553  is_committed = true;
554  is_insync = false;
555 
556  if(hw && !hw->disarm())
557  sync();
558  }
559 
560  // clear residual error (if any)
561  last_err = "";
562  scanIoRequest(onErr);
563 
564  scanIoRequest(changed);
565  DEBUG(1, ("Committed\n") );
566 }

609 // Called from ISR context
610 void SoftSequence::sync()
611 {
612  DEBUG(3, ("Syncing %c\n", is_insync ? 'Y' : 'N') );
613  if(is_insync)
614  {DEBUG(3, ("Skip\n")); return;}
615 
616  assert(hw);
617 
618  if(nat_ioread32(hw->ctrlreg)&EVG_SEQ_RAM_RUNNING) {
619  // we may still be _ENABLED at this point, but the trigger source is set to
620  // Disabled, so this makes no difference.
621  epicsInterruptContextMessage("SoftSequence::sync() while running\n");
622  return;
623  }
624 
625  // At this point the sequencer is not running and effectively disabled.
626  // From paranoia, reset it anyway
627  nat_iowrite32(hw->ctrlreg, hw->ctrlreg_hw | EVG_SEQ_RAM_RESET);
628 
629  hw->ctrlreg_user &= ~(EVG_SEQ_RAM_REPEAT_MASK|EVG_SEQ_RAM_SRC_MASK);
630 
631  switch(committed.mode) {
632  case Single:
633  hw->ctrlreg_user |= EVG_SEQ_RAM_SINGLE;
634  break;
635  case Normal:
636  hw->ctrlreg_user |= EVG_SEQ_RAM_NORMAL;
637  break;
638  }
639 
640  epicsUInt8 src;
641 
642  // default to disabled
643  switch(owner->type) {
644  case SeqManager::TypeEVG:
645  src = 31;
646  break;
647  case SeqManager::TypeEVR:
648  src = 63;
649  break;
650  default:
651  return;
652  }
653 
654  // paranoia: disable any external trigger mappings
655  owner->mapTriggerSrc(hw->idx, 0x02000000);
656 
657  // map trigger source codes
658  // MSB governs the type of mapping
659  switch(committed.src&0xff000000) {
660  case 0x00000000: // raw mapping
661  DEBUG(5, (" Raw mapping %x\n", committed.src));
662  // LSB is code
663  src = committed.src&0xff;
664  break;
665  case 0x01000000: // software trigger mapping
666  DEBUG(5, (" SW mapping %x\n", committed.src));
667  // ignore 0x00ffffff
668  switch(owner->type) {
669  case SeqManager::TypeEVG:
670  src = 17+hw->idx;
671  break;
672  case SeqManager::TypeEVR:
673  src = 61;
674  break;
675  }
676  break;
677  case 0x02000000: // external trigger
678  DEBUG(5, (" EXT mapping %x\n", committed.src));
679  if(owner->type==SeqManager::TypeEVG) {
680  // pass through to sub-class
681  owner->mapTriggerSrc(hw->idx, committed.src);
682  src = 24+hw->idx;
683  }
684  break;
685  case 0x03000000: // disable trigger
686  DEBUG(5, (" NO mapping %x\n", committed.src));
687  // use default
688  break;
689  default:
690  DEBUG(0, ("unknown sequencer trigger code %08x\n", (unsigned)committed.src));
691  break;
692  }
693  DEBUG(5, (" Trig Src %x\n", src));
694 
695  hw->ctrlreg_user |= src;
696 
697  // write out the RAM
698  volatile epicsUInt32 *ram = static_cast<volatile epicsUInt32 *>(hw->rambase);
699  for(size_t i=0, N=committed.codes.size(); i<N; i++)
700  {
701  nat_iowrite32(ram++, committed.times[i]);
702  nat_iowrite32(ram++, committed.codes[i]);
703  if(committed.codes[i]==0x7f)
704  break;
705  }
706 
707  {
708  epicsUInt32 ctrl = hw->ctrlreg_hw = hw->ctrlreg_user;
709  if(is_enabled)
710  ctrl |= EVG_SEQ_RAM_ARM;
711  else
712  ctrl |= EVG_SEQ_RAM_DISABLE; // paranoia
713 
714  DEBUG(3, (" SeqCtrl %x\n", ctrl));
715  nat_iowrite32(hw->ctrlreg, ctrl);
716  }
717 
718  is_insync = true;
719  DEBUG(3, ("In Sync\n") );
720 }
