62 std::string EVR::position() const
63 {
64  std::ostringstream position;
65 
66  if(busConfiguration.busType == busType_pci) {
67  position << std::hex << busConfiguration.pci.dev->bus << ":"
68  << std::hex << busConfiguration.pci.dev->device << "."
69  << std::hex << busConfiguration.pci.dev->function;
70  if(busConfiguration.pci.dev->slot)
71  position << " slot=" << busConfiguration.pci.dev->slot;
72  } else if(busConfiguration.busType == busType_vme) {
73  position << "Slot #" << busConfiguration.vme.slot;
74  } else {
75  position << "Unknown position";
76  }
77 
78  return position.str();
79 }
