100 MRMOutput::sourceInternal() const
101 {
102  epicsUInt32 val=64; // an invalid value
103  switch(type) {
104  case OutputInt:
105  return READ32(owner->base, IRQPulseMap) & 0xffff;
106  case OutputFP:
107  val = READ32(owner->base, OutputMapFP(N)); break;
108  case OutputFPUniv:
109  val = READ32(owner->base, OutputMapFPUniv(N)); break;
110  case OutputRB:
111  val = READ32(owner->base, OutputMapRB(N)); break;
112  case OutputBackplane:
113  val = READ32(owner->base, OutputMapBackplane(N)); break;
114  }
115  val &= Output_mask(N);
116  val >>= Output_shift(N);
117  return val;
118 }
