26 MRMCML::MRMCML(const std::string& n, unsigned char i,EVRMRM& o, outkind k, formFactor f)
27  :CML(n)
28  // CML word length
29  ,mult(f==formFactor_CPCIFULL || f==formFactor_mTCA ? 40 : 20)
30  // # of 32-bit dwords used to store 1 CML word
31  // 40 bits fit in 2 dwords, 20 bits fit in 1
32  ,wordlen(f==formFactor_CPCIFULL || f==formFactor_mTCA ? 2 : 1)
33  ,base(o.base)
34  ,N(i)
35  ,owner(o)
36  ,shadowEnable(0)
37  ,shadowWaveformlength(0)
38  ,kind(k)
39 {
40  epicsUInt32 val=READ32(base, OutputCMLEna(N));
41 
42  val&=~OutputCMLEna_type_mask;
43 
44  switch(kind) {
45  case typeCML: break;
46  case typeTG203: val|=OutputCMLEna_type_203; break;
47  case typeTG300: val|=OutputCMLEna_type_300; break;
48  default:
49  throw std::invalid_argument("Invalid CML kind");
50  }
51 
52  for(size_t i=0; i<NELEMENTS(shadowPattern); i++) {
53  epicsUInt32 L = wordlen * (lenPatternMax((pattern)i)/mult);
54  shadowPattern[i] = new epicsUInt32[L];
55  std::fill(shadowPattern[i], shadowPattern[i]+L, 0);
56  }
57 
58  shadowEnable=val;
59 }

85 void
86 MRMCML::setMode(cmlMode m)
87 {
88  epicsUInt32 mask=0;
89  switch(m) {
90  case cmlModeOrig: mask |= OutputCMLEna_mode_orig; break;
91  case cmlModeFreq: mask |= OutputCMLEna_mode_freq; break;
92  case cmlModePattern: mask |= OutputCMLEna_mode_patt; break;
93  default:
94  throw std::out_of_range("Invalid CML Mode");
95  }
96  bool wasenabled = enabled();
97  shadowEnable &= ~OutputCMLEna_ena; // disable while syncing
98  shadowEnable &= ~OutputCMLEna_mode_mask;
99  shadowEnable |= mask;
100  WRITE32(base, OutputCMLEna(N), shadowEnable);
101 
102 
103  switch(m) {
104  case cmlModeOrig:
105  WRITE32(base, OutputCMLPatLength(N), 0);
106  syncPattern(patternFall);
107  syncPattern(patternHigh);
108  syncPattern(patternLow);
109  syncPattern(patternRise);
110  break;
111 
112  case cmlModePattern:
113  WRITE32(base, OutputCMLPatLength(N), shadowWaveformlength-1);
114  syncPattern(patternWaveform);
115  break;
116 
117  default:
118  break;
119  }
120 
121  if(wasenabled)
122  shadowEnable |= OutputCMLEna_ena; // enable after syncing
123  WRITE32(base, OutputCMLEna(N), shadowEnable);
124 }

351 epicsUInt32
352 MRMCML::getPattern(pattern p, unsigned char *buf, epicsUInt32 blen) const
353 {
354  epicsUInt32 plen=lenPattern(p);
355 
356  // number of bytes of 'buf' to fill
357  blen = std::min(plen, blen);
358 
359  epicsUInt32 val=0;
360  for(epicsUInt32 i=0; i<blen; i++) {
361  size_t cmlword = (i/mult);
362  size_t cmlbit = (i%mult);
363 
364  size_t cpuword, cpubit;
365  bool first; // first bit in CPU word
366 
367  if(mult<32) {
368  first = cmlbit==0;
369  cpuword = cmlword;
370  cpubit = 19 - cmlbit;
371  } else {
372  first = cmlbit==0 || cmlbit==8;
373  cpuword = 2*cmlword + (cmlbit<8 ? 0 : 1);
374  cpubit = cmlbit<8 ? 7-cmlbit : 31-(cmlbit-8);
375  }
376 
377  if(first) {
378  val=shadowPattern[p][cpuword];
379  }
380 
381  buf[i]=val>>cpubit;
382  buf[i]&=0x1;
383 
384  }
385  return blen;
386 }

388 void
389 MRMCML::setPattern(pattern p, const unsigned char *buf, epicsUInt32 blen)
390 {
391  // If we are given a length that is not a multiple of CML word size
392  // then truncate.
393  if(blen%mult){
394  printf("Given length %u is not a multiple of %u (CML word size). Truncating...\n", blen, mult);
395  blen-=blen%mult;
396  }
397 
398  if(blen>lenPatternMax(p))
399  throw std::out_of_range("Pattern is too long");
400 
401 
402  epicsUInt32 val=0;
403  for(epicsUInt32 i=0; i<blen; i++) {
404  size_t cmlword = (i/mult);
405  size_t cmlbit = (i%mult);
406  size_t cpuword, cpubit;
407  if(mult<32) {
408  cpuword = cmlword;
409  cpubit = 19 - cmlbit;
410  } else {
411  cpuword = 2*cmlword + (cmlbit<8 ? 0 : 1);
412  cpubit = cmlbit<8 ? 7-cmlbit : 31-(cmlbit-8);
413  }
414 
415  val|=(!!buf[i])<<cpubit;
416 
417  // cpubit is counting down.
418  // write complete dword after the last bit is set
419  if(cpubit==0) {
420  shadowPattern[p][cpuword] = val;
421  val=0;
422  }
423  }
424 
425  if(p==patternWaveform)
426  shadowWaveformlength = blen/mult;
427 
428  // temporarly disable when changing multi dword patterns
429  // to prevent output of incomplete patterns
430  bool active=enabled();
431  if(active)
432  enable(false);
433 
434  if(mode()==cmlModePattern)
435  WRITE32(base, OutputCMLPatLength(N), shadowWaveformlength-1);
436 
437  syncPattern(p);
438 
439  if(active)
440  enable(true);
441 }

443 void
444 MRMCML::syncPattern(pattern p)
445 {
446  if(mult==20 && p!=patternWaveform) {
447  // for 20 bit patterns modifying the 4x pattern
448  // can be done without effecting the waveform pattern
449  switch(p) {
450  case patternHigh:
451  WRITE32(base, OutputCMLHigh(N), shadowPattern[patternHigh][0]); return;
452  case patternFall:
453  WRITE32(base, OutputCMLFall(N), shadowPattern[patternFall][0]); return;
454  case patternLow:
455  WRITE32(base, OutputCMLLow(N), shadowPattern[patternLow][0]); return;
456  case patternRise:
457  WRITE32(base, OutputCMLRise(N), shadowPattern[patternRise][0]); return;
458  default:
459  throw std::logic_error("syncPattern: invalid state 20");
460  }
461  }
462 
463  cmlMode curmode=mode();
464  if(curmode==cmlModeFreq)
465  return; // Don't know which pattern to sync...
466 
467  switch(curmode) {
468  case cmlModeOrig:
469 
470  switch(p) {
471  case patternLow:
472  WRITE32(base, OutputCMLPat(N, 0), shadowPattern[patternLow][0]);
473  WRITE32(base, OutputCMLPat(N, 1), shadowPattern[patternLow][1]);
474  break;
475 
476  case patternRise:
477  WRITE32(base, OutputCMLPat(N, 2), shadowPattern[patternRise][0]);
478  WRITE32(base, OutputCMLPat(N, 3), shadowPattern[patternRise][1]);
479  break;
480 
481  case patternFall:
482  WRITE32(base, OutputCMLPat(N, 4), shadowPattern[patternFall][0]);
483  WRITE32(base, OutputCMLPat(N, 5), shadowPattern[patternFall][1]);
484  break;
485 
486  case patternHigh:
487  WRITE32(base, OutputCMLPat(N, 6), shadowPattern[patternHigh][0]);
488  WRITE32(base, OutputCMLPat(N, 7), shadowPattern[patternHigh][1]);
489  break;
490 
491  case patternWaveform:
492  break; // not safe to sync
493  }
494  break;
495 
496  case cmlModePattern:
497  switch(p) {
498  case patternWaveform:
499  for(size_t i=0; i<shadowWaveformlength*wordlen; i++)
500  WRITE32(base, OutputCMLPat(N, i), shadowPattern[patternWaveform][i]);
501  break;
502  default:
503  break; // not safe to sync
504  }
505 
506  break;
507 
508  default:
509  throw std::logic_error("syncPattern: invalid state 40");
510  }
